CREATE TABLE MonHoc(MaMH VARCHAR(40) PRIMARY KEY, TenMH VARCHAR(40) NOT NULL, SoTC INT NOT NULL);



CREATE TABLE GiaoVien(MaGV VARCHAR(40) PRIMARY KEY, HoGV VARCHAR(40) NOT NULL, TenGV VARCHAR(40) NOT NULL, DonVi VARCHAR(40) NOT NULL);



CREATE TABLE SinhVien(MaSV VARCHAR(40) PRIMARY KEY, HoSV VARCHAR(40) NOT NULL, TenSV VARCHAR(40) NOT NULL, NgaySinh DATE NOT NULL, NoiSinh VARCHAR(40) NOT NULL);

CREATE TABLE Lop(MaLop VARCHAR(40) PRIMARY KEY, MaMH VARCHAR(40) NOT NULL, HocKy INT NOT NULL, NamHoc INT NOT NULL, MaGV VARCHAR(40) NOT NULL, FOREIGN KEY (MaMH) REFERENCES MonHoc(MaMH), FOREIGN KEY (MaGV) REFERENCES GiaoVien(MaGV));


CREATE TABLE SinhVienLop(MaSV VARCHAR(40), MaLop VARCHAR(40), Diem DOUBLE NOT NULL, PRIMARY KEY(MaSV, MaLop), FOREIGN KEY (MaSV) REFERENCES SinhVien(MaSV), FOREIGN KEY (MaLop) REFERENCES Lop(MaLop));
